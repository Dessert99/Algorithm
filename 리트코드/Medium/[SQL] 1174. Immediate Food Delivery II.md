```sql
WITH
    -- id 별로 나눠서 주문 일자 기준 정렬한 뒤, 숫자를 매긴 컬럼 추가한 테이블
    result AS (
        SELECT
            customer_id,
            order_date,
            customer_pref_delivery_date,
            ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date) AS rnk
        FROM
            Delivery 
    )

SELECT
    ROUND((
    100 * ( SUM(CASE WHEN order_date=customer_pref_delivery_date THEN 1 ELSE 0 END) / COUNT(customer_id))
    )
    ,2) AS immediate_percentage

FROM
    result
WHERE
    -- 최초 주문 정보만 가져온다.
    rnk = 1
```
