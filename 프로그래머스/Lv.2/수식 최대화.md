```javascript
function solution(expression) {
    
    // 숫자, 연산을 각각 담는다.
    const numArr= []
    const opArr = []
    
    // 숫자 시작 부분
    let left = 0
    let right = 0

    while(right<expression.length){
        // 숫자인가 연산자인가?
        const cur = expression[right]
        // 연산자를 만났다면
        if(cur === '+' || cur === '-'|| cur==='*'){
            // left ~ right직전까지는 숫자이다.
            numArr.push(Number(expression.slice(left, right)))
            opArr.push(cur)
            
            //포인터 초기화
            right++
            left = right
        }else{
            // 현재 위치가 숫자라면
            right++ // 오른쪽 인덱스만 옮긴다.
        }
    }
    // 마지막 left~끝까지는 숫자
    numArr.push(Number(expression.slice(left)))

    // 결과 배열에 담기
    const result = []
    // 연산 개수 기준으로 담고, 마지막에 숫자를 넣는다.
    for(let i =0;i<opArr.length;i++){
        result.push(numArr[i])
        result.push(opArr[i])
    }
    result.push(numArr[numArr.length-1]) // 완성
    
    // 조합
    const comb = [
        ['*','-','+'],['*','+','-'],
        ['+','-','*'],['+','*','-'],
        ['-','*', '+'],['-','+','*']
    ]
    
    let answer = Number.MIN_SAFE_INTEGER
    // 모든 조합 확인
    for (const c of comb){
        // 복사본
        const copy = [...result]
        
        // 현재 조합에서 결과
        
        
        // c조합 반복
        for (const op of c){
            // op = 현재 연산 -,+,* 중 하나
            
            // splice로 배열이 계속 바뀌기 때문에, indexOf로 접근해야 한다.
            while(copy.includes(op)) {
                // 해당 연산의 인덱스
                const j = copy.indexOf(op)
                
                let r; // 연산이 끝난 결과
                switch (op){
                    case '-':
                        r = copy[j-1] - copy[j+1]
                        break
                    case '+':
                        r = copy[j-1] + copy[j+1]
                        break
                    case '*':
                        r = copy[j-1] * copy[j+1]
                        break
                }
                    // 연산 결과 r을 올바른 위치에 넣는다.
                    copy.splice(j-1, 3, r) // 인덱스 j위치가 연산이니까, 그 전부터 3개까지 값을 연산 결과인 r로 바꾼다.
            }
        }
        // 결과 갱신
        answer = Math.max(answer, Math.abs(copy[0]))
        
    }
    return answer;
}
```
### 🔗 풀이
1. 🤔 일단 최대 6가지 경우의 수를 해봐야할 것 같다.
    - 아무리 그리디라고 해도 계산을 안해보고 최댓값이 나오는지 모를 것 같다

2. 👀 방법1) 
    1. 일단 6가지 경우의 수를 전부 해본다.
    2. 만약 연산이 하나 없더라도, 그냥 넘어가면 된다.
    3. 만약 - > * > + 순서라면 -부터 

3. 🤔 expression을 어떻게 배열로 만들까?
    - 포인터 사용
    - 숫자, 연산 배열에 각각 담아서 나중에 합침

4. 🤔 우선순위 조합을 만들어서 전부 구해보기
    - 6가지라서 괜찮다.
    
5. 🤔 계산 전략
    1. 복사본 copy 준비
    2. 어떤 조합 c = ['-', '+', '*'] 대상
    3. copy를 순회하며 현재 c원소 연산을 모두 계산
    4. 모든 조합 반복
    5. 이제 남은 숫자에 절댓값 걸고 최댓값 비교

### 🔗 배운점
1. 주어진 문자열을 부호 종류에 맞게 파싱하는 방법
   - 원래 이런 문제는 배열로 주어졌다. 하지만 이번에는 하나의 문자열로 제공되며, 부호 종류게 따라 파싱하는 과정이 필요했다.
   - 나는 포인터를 활용해서 숫자, 연산 배열에 넣어놓고 나중에 합치는 방법으로 파싱하였다.
2. `+`, `-`, `*` 모든 조합을 확인해야 한다.
   - 이 문제는 그리디로 풀 수 없다. 어떤 연산이 최선인지 모르기 때문에 모든 조합을 시도해봐야 했다. 어차피 6가지밖에 없다.
   - `comb`라는 배열에 6가지 조합을 놓고 순서대로 연산했다. 문제가 길어질수록 이렇게 case를 만들어놔야 하는 상황이 발생한다.
   - 이때, `copy`라는 복사본을 만들어야 한다. 나중에 사용할 `splice`메서드는 원본 배열을 변경하기 때문이다.
3. 우선순위에 따라 연산하기
   - 나는 주어진 연산 조합을 앞에서부터 차례로 계산하는 방식으로 풀었다. 배열의 길이가 작아서 가능했다.
   - `while` + `includes`로 현재 연산이 사라질 때 까지 반복했다. 이때 `for`문을 사용하지 못하는 이유는, `splice`가 원본 배열을 파괴하기 때문이다.
   - 인덱스는 그때마다 `indexOf()`로 구해야 한다.
   - 만약 주어진 연산`-`의 인덱스가 2라고 할 때, 숫자는 인덱스 1, 3번에 위치한다. 결과를 변수에 저장하고 `splice`로 1번 인덱스부터 3번인덱스를 지운 뒤, 결과를 넣으면 된다.
