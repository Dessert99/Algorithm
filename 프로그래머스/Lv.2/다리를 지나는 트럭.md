class Queue {
    constructor() {
        this.store = [];
        this.front = 0;
        this.rear = 0;
    }
    size() {
        return this.rear - this.front;
    }

    enqueue(value) {
        this.store[this.rear] = value;
        this.rear++;
    }

    dequeue() {
        if (this.front === this.rear) {
            return null;
        }
        const value = this.store[this.front];
        this.front++;
        return value;
    }
    // ë§¨ ì• íŠ¸ëŸ­ í™•ì¸ìš©
    peek() {
        return this.store[this.front];
    }
}
