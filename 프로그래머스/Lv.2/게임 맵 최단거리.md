function solution(maps) {
    var answer = 0;
    // n: μ„Έλ΅ m: κ°€λ΅
    const m = maps[0].length
    const n = maps.length

    
    const dx = [-1, 0, 1, 0];
    const dy = [0, -1, 0, 1];
    
    // ν μƒμ„±
    class Queue {
        constructor(){
            this.store = [];
            this.front = 0;
            this.rear = 0
        }
    
        size() {
        return this.rear - this.front;
    }
        enqueue(value){
            this.store[this.rear] = value;
            this.rear++
        }
        
        dequeue(){
            if (this.front===this.rear){
                return undefined
            }
            const value = this.store[this.front]
            this.front++
            return value
        }
    }
    
    const q = new Queue();
    
    // μ΄κΈ°μ„Έν…
    q.enqueue([0,0])
    
    // λ°λ³µλ¬Έ μ‹μ‘
    while(q.size() > 0){
        const [curY, curX] = q.dequeue()
        
        // ν„μ¬ μ„μΉμ—μ„ μƒν•μΆμ° μ„μΉλ΅ κ° μ μλ”μ§€ νλ‹¨
        for (let i = 0; i<4; i++){
            const ny = curY + dy[i]
            const nx = curX + dx[i]
            
            // νƒμƒ‰ μ΅°κ±΄ 1) μ ν¨ν• λ²”μ„ 2) λ°©λ¬Έν•μ§€ μ•μ€ κ³³
            if (ny>=0 && nx>=0 && ny<n && nx<m && maps[ny][nx] === 1 ){
                // ν„μ¬ μ„μΉμ—μ„ + 1
                maps[ny][nx] = maps[curY][curX] + 1
                // νμ— λ„£κΈ°
                q.enqueue([ny, nx])
            }
        }
    }
    
    answer = maps[n-1][m-1] === 1 ? -1 : maps[n-1][m-1]
    return answer;
}
