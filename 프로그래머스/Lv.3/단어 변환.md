class Queue {
    constructor() {
        this.store = [];
        this.front = 0;
        this.rear = 0;
    }

    size() {
        return this.rear - this.front;
    }

    // λ„£κΈ°
    enqueue(value) {
        this.store[this.rear] = value;
        this.rear++;
    }
    // λΉΌκΈ°
    dequeue() {
        if (this.front === this.rear) {
            return undefined;
        }
        const value = this.store[this.front];
        this.front++;
        return value;
    }
}
