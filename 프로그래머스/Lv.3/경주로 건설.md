class MinHeap {
    constructor(){
        this.heap = []
    }
    swap(i1,i2){
        [this.heap[i1], this.heap[i2]] = [this.heap[i2], this.heap[i1]]
    }
    getParentIndex(curIndex){
        return Math.floor((curIndex-1)/2)
    }
    getLeftChildIndex(curIndex){
        return curIndex * 2 + 1
    }
    getRightChildIndex(curIndex){
        return curIndex *2 + 2
    }
    
    insert(value){
        this.heap.push(value)
        
        this.heapifyUp()
    }
    heapifyUp(){
        let curIndex = this.heap.length-1
        
        while(curIndex>0){
            const parentIndex = this.getParentIndex(curIndex)
            
            if (this.heap[parentIndex][3] <= this.heap[curIndex][3]){
                break
            }
            
            this.swap(curIndex, parentIndex)
            curIndex = parentIndex
        }
    }
    
    extractMin(){
        if (this.heap.length===0){
            return null
        }
        if (this.heap.length===1){
            return this.heap.pop()
        }
        
        const min = this.heap[0]
        this.heap[0] = this.heap.pop()
        
        this.heapifyDown()
        
        return min
    }
    heapifyDown(){
        let curIndex =0
        const n = this.heap.length
        
        while(true){
            let minIndex= curIndex
            const leftChildIndex = this.getLeftChildIndex(curIndex)
            const rightChildIndex = this.getRightChildIndex(curIndex)
            
            if (leftChildIndex< n && this.heap[leftChildIndex][3] < this.heap[minIndex][3]){
                minIndex = leftChildIndex
            }
            if (rightChildIndex< n && this.heap[rightChildIndex][3] < this.heap[minIndex][3]){
                minIndex = rightChildIndex
            }
            if (curIndex===minIndex){
                break
            }
            
            this.swap(curIndex,minIndex)
            curIndex= minIndex
        }
    }
}

function solution(board) {
    let answer = 0;
    // 방향 정보: 
    const dir = {
        0: [-1, 0], // 위쪽으로 감
        1: [0,1], // 오른쪽으로 감
        2: [1,0], // 아래쪽으로 감
        3: [0,-1] // 왼쪽으로 감
    }
    
    //보드 길이
    const N = board.length
    
    //가중치 배열
    const dist = Array.from({length:N},()=>Array.from({length:N}, ()=>Array(4).fill(Infinity)))
    dist[0][0][0] = 0 // 출발지 초기화
    
    const pq = new MinHeap()
    // (0,0)에서 출발한 값 넣기
    for (let i=0;i<4;i++){
        const [y,x] = dir[i]
        
        if (y>=0 && x>=0&& y<N&& x<N&& board[y][x] ===0){
            // [y,x,방향, 가중치]
            pq.insert([y,x,i,100])
            dist[y][x][i] = 100
        }
    }
    
    // 이제 가중치 탐색 시작
    while(pq.heap.length>0){
        // 현재 위치 정보
        const [curY,curX,curDir,curCost] = pq.extractMin()
        
        // 이미 최적화된 경로가 있다면
        if (dist[curY][curX][curDir] < curCost){
            continue
        }
        
        // 4방향 확인
        for (let i=0;i<4;i++){
            const ny = curY + dir[i][0]
            const nx = curX + dir[i][1]
            
            const nextDir = i
            
            // 일단 유효한 범위인지?
            if (ny>=0&&nx>=0&&ny<N&&nx<N&&board[ny][nx]===0){
                // 방향에 따라 다음 위치 가중치 결정
                let nextCost;
                if (curDir === nextDir){
                    // 방향이 같다면 100
                    nextCost = curCost + 100
                }else{
                    //방향이 다르다면 500
                    nextCost = curCost + 600
                }
                
                // 현재 저장된 가중치보다 효율적일 때만 탐색
                if (nextCost<dist[ny][nx][nextDir]){
                    dist[ny][nx][nextDir] = nextCost
                    pq.insert([ny,nx,nextDir,nextCost])
                }
            }
        }
    }
    
    // 도착지 4방향 중 최솟값
    answer = Math.min(...dist[N-1][N-1])
    
    return answer;
}
