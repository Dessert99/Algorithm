[https://www.acmicpc.net/problem/20366](https://www.acmicpc.net/problem/20366)
```javascript
const input = require('fs')
    .readFileSync(process.platform === 'linux' ? '/dev/stdin' : './input.txt')
    .toString()
    .trim()
    .split('\n');

let inputIndex = 0;

// 4~600
const N = Number(input[inputIndex++]);

// 눈덩이의 지름들
const snowH = input[inputIndex++].split(' ').map(Number);

// 눈사람 차이
let answer = Number.MAX_SAFE_INTEGER;

snowH.sort((a, b) => a - b);

const tmp = [];

// 조합 -> ❌ 이건 사실상 100% 맞는 풀이는 아니다. 백트래킹보다 3중 반복문이 더 적절한 풀이.
function backtracking(cnt, startIndex) {
    // 기저 조건
    if (cnt === 2) {
        // tmp 안에 있는 인덱스들은 첫 번째 눈사람을 만들기 위한 눈덩이 인덱스이다.
        secondSnowman(tmp[0], tmp[1]);

        return;
    }

    // 기본 조건
    for (let i = startIndex; i < N; i++) {
        // 넣고(인덱스를 넣는다.)
        tmp.push(i);
        //재귀
        backtracking(cnt + 1, i + 1);
        //빼기
        tmp.pop();
    }
}

function secondSnowman(idx1, idx2) {
    // idx1, idx2 는 빼고 투 포인터
    let left = 0;
    let right = snowH.length - 1;

    // 첫 번째 눈사람 길이
    const firstHeight = snowH[idx1] + snowH[idx2];

    while (left < right) {
        if (left === idx1 || left === idx2) {
            left++;
            continue;
        }
        if (right === idx1 || right === idx2) {
            right--;
            continue;
        }

        // 두 번째 눈사람 길이
        const secondHeight = snowH[left] + snowH[right];

        // 두 눈사람의 차이
        const gap = firstHeight - secondHeight;

        // 차이 갱신
        answer = Math.min(answer, Math.abs(gap));

        if (gap === 0) {
            // 차이가 0이면 더 볼 필요가 없다.
            break;
        } else if (gap > 0) {
            // 차이가 양수라면? 두 번째 눈사람을 조금 크게 만든다.
            left++;
        } else {
            // 차이가 음수라면? 두 번째 눈사람을 작게 만든다.
            right--;
        }
    }
}
backtracking(0, 0);

console.log(answer);
```
### 🔗 사고 과정
1. 💡 문제 분석
    - 5개 눈덩이가 있고, 각 눈덩이는 지름이 2,3,4..등이다.
    - 눈사람 키 차이가 최소이어야 한다.
        1. 4개의 눈덩이 조합 중 2,2로 나누었을 때
        2. 그 차이가 최소이어야 한다.

2. 🤔 조합?
    - ✅ 일단 모든 경우의 수를 찾아봐야 한다. 근데 600개 중 4개를 뽑는 조합은 시간복잡도 초과이다.
    - 3중 반복문까지도 안전하다. -> ✅ 사실 이걸로 풀었어야 했다.
    - 600개 중 2개를 집는 방식은?
        - 약 18만이다.
    - 일단, 두 개의 눈덩이를 집어서 눈사람을 만든 뒤에, 나머지 2개는 투 포인터로 한 번의 순회로 끝내면 시간 안에 끝낼 수 있다. 대략 1억번. 문제는 2초가 주어짐.

3. 🤔 정렬해야 하는가?
    - ✅ 조합을 하지 못하는 경우, 정렬로 어느 정도 규칙성을 부여하는게 좋아보인다.
### 🔗 배운점
1. 정렬을 해야 하는 이유
   - 이 문제같은 경우는 눈사람 차이를 늘리고, 줄여가며 비교해야 해서 "연속성"이 필요하다. 이것이 정렬이 필요한 이유이다.
2. 백트래킹보다 3중 반복문이 더 적절한 풀이이다.
   - 평소에 조합 === 백트래킹으로 생각하고 있어서 다른 방법을 고려하지 않았다. 백트래킹을 할 경우 메모리 초과 발생 위험이 있다.
3. `for`문으로 조합 만드는 방법
   > 중첩 횟수 = 뽑는 개수
   - 예시
     ```javascript
        const items = ['A', 'B', 'C', 'D', 'E'];
        const n = items.length;
        
        for (let i = 0; i < n; i++) {
            for (let j = i + 1; j < n; j++) {
                console.log(`${items[i]}, ${items[j]}`);
            }
        }
       /*
            A, B
            A, C
            A, D
            A, E
            B, C
            B, D
            B, E
            C, D
            C, E
            D, E
       */
     ```
