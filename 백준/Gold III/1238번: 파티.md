```javascript
const input = require('fs')
    .readFileSync(process.platform === 'linux' ? '/dev/stdin' : './input.txt')
    .toString()
    .trim()
    .split('\n');

let inputIndex = 0;

class MinHeap {
    constructor() {
        this.heap = [];
    }
    swap(i1, i2) {
        [this.heap[i1], this.heap[i2]] = [this.heap[i2], this.heap[i1]];
    }
    getParentIndex(curIndex) {
        return Math.floor((curIndex - 1) / 2);
    }
    getLeftChildIndex(curIndex) {
        return curIndex * 2 + 1;
    }
    getRightChildIndex(curIndex) {
        return curIndex * 2 + 2;
    }

    insert(value) {
        this.heap.push(value);
        this.heapifyUp();
    }
    heapifyUp() {
        let curIndex = this.heap.length - 1;

        while (curIndex > 0) {
            const parentIndex = this.getParentIndex(curIndex);

            if (this.heap[parentIndex][1] <= this.heap[curIndex][1]) {
                break;
            }

            this.swap(curIndex, parentIndex);
            curIndex = parentIndex;
        }
    }

    extractMin() {
        if (this.heap.length === 0) {
            return null;
        }
        if (this.heap.length === 1) {
            return this.heap.pop();
        }
        const min = this.heap[0];
        this.heap[0] = this.heap.pop();

        this.heapifyDown();

        return min;
    }
    heapifyDown() {
        let curIndex = 0;
        const n = this.heap.length;

        while (true) {
            let minIndex = curIndex;
            const leftChildIndex = this.getLeftChildIndex(curIndex);
            const rightChildIndex = this.getRightChildIndex(curIndex);

            if (leftChildIndex < n && this.heap[leftChildIndex][1] < this.heap[minIndex][1]) {
                minIndex = leftChildIndex;
            }
            if (rightChildIndex < n && this.heap[rightChildIndex][1] < this.heap[minIndex][1]) {
                minIndex = rightChildIndex;
            }
            if (curIndex === minIndex) {
                break;
            }

            this.swap(curIndex, minIndex);
            curIndex = minIndex;
        }
    }
}

// N개의 마을(사람)과 M개의 도로. X마을에서 파티
const [N, M, X] = input[inputIndex++].split(' ').map(Number);

// 기본 마을 구조. X마을에서 각 마을로 "복귀"할 때 걸리는 비용을 저장한다.
const adjList = Array.from({ length: N + 1 }, () => []);

// 도로 방향을 거꾸로 한 마을 구조. 시작 지점은 X지만, 각 마을에서 X까지의 최단 거리를 의미한다.
const reversedAdjList = Array.from({ length: N + 1 }, () => []);

for (let i = 1; i <= M; i++) {
    // i번째 도로의 [시작 마을, 도착 마을, 소요 시간]
    const [start, end, time] = input[inputIndex++].split(' ').map(Number);

    adjList[start].push([end, time]);
    reversedAdjList[end].push([start, time]);
}

// 가중치 배열: X마을에서 각 마을로 복귀 시간을 저장한 배열
const dist = Array.from({ length: N + 1 }, () => Infinity);
dist[X] = 0; // X마을이 출발지

// 거꾸로된 도로 가중치 배열: 결과물은 각 마을에서 X까지의 최단 비용이다.
const reversedDist = Array.from({ length: N + 1 }, () => Infinity);
reversedDist[X] = 0; // 여전히 출발지는 X

const pq = new MinHeap();
pq.insert([X, 0]); // 초기화

// 복귀하는 시간 구하기
while (pq.heap.length > 0) {
    // 지금 확인할 마을
    const [curNode, curCost] = pq.extractMin();

    // 이미 최단 거리가 저장되었다면 패스한다.
    if (dist[curNode] < curCost) {
        continue;
    }

    // 지금 마을과 연결된 마을들
    const linked = adjList[curNode];

    // 연결된 마을들 확인
    for (let i = 0; i < linked.length; i++) {
        const [nextNode, cost] = linked[i];

        // 이동한다면 걸리는 비용
        const nextCost = curCost + cost;

        // 이동할 노드의 누적 최단거리보다, 지금 이동할 때 걸리는 비용이 더 효율적일 때만 이동한다.
        if (dist[nextNode] > nextCost) {
            dist[nextNode] = nextCost;
            pq.insert([nextNode, nextCost]);
        }
    }
}
// 이제 거꾸로 연결된 지도를 확인한다.
pq.insert([X, 0]);
while (pq.heap.length > 0) {
    // 지금 확인할 마을
    const [curNode, curCost] = pq.extractMin();

    // 이미 최단 거리가 저장되었다면 패스한다.
    if (reversedDist[curNode] < curCost) {
        continue;
    }

    // 지금 마을과 연결된 마을들
    const linked = reversedAdjList[curNode];

    // 연결된 마을들 확인
    for (let i = 0; i < linked.length; i++) {
        const [nextNode, cost] = linked[i];

        // 이동한다면 걸리는 비용
        const nextCost = curCost + cost;

        // 이동할 노드의 누적 최단거리보다, 지금 이동할 때 걸리는 비용이 더 효율적일 때만 이동한다.
        if (reversedDist[nextNode] > nextCost) {
            reversedDist[nextNode] = nextCost;
            pq.insert([nextNode, nextCost]);
        }
    }
}
// 두 가중치 배열을 더한다.
const answer = [];

for (let i = 1; i < dist.length; i++) {
    answer.push(dist[i] + reversedDist[i]);
}
console.log(Math.max(...answer));
```
### 🔗 풀이
1. 💡 문제 파악
    1. 5개의 마을에 5명의 학생이 3번째 마을에서 파티함
    2. 이 마을에는 10개의 도로가 존재
    3. 2번째 길을 지나는데 70시간 소비
    4. 도로들은 단방향이기 때문에 아마 그들이 오고 가는 길이 다를지도 모른다 -> 다익스트라 2번 해야할수도?

2. 🤔 다익스트라 포인트
    1. ✅ 가중치
    2. ✅ 최단거리

3. 🤔 가장 오래 걸리는 학생?
    1. 학생은 1~N 마을에 위치한다.
    2. N개의 마을에서 X까지, 다시 복귀하는 계산을 한다.
    3. X부터 각자 마을로 복귀하는 로직은 한 번만 하면 된다. 총 N+1번 다익스트라 하면 된다.
    4. N은 최대 1000개, 도로는 최대 10000개이다. 다익스트라 시간복잡도는 O(ElogV)로, 11000 * 1000 = 1천만 정도 된다.
    5. ✅ 이걸 한 번에 처리히는 방법은 없을까?

4. 🤔 N번의 다익스트라를 한번에 처리하는 방법?
    1. 보통 다익스트라를 푸는 방식을 정리해보자.
        - 어떤 출발지가 있고 도착지까지 걸리는 최단거리를 dist배열에 저장한다.
        - dist배열은, 어떤 A라는 노드에서 각 노드까지의 최단 거리를 누적으로 저장한 것이다.
    2. 지금 상황은 출발 지점이 여러 곳이다.
        -> ✅ 출발 지점을 하나로 만들어야 한다. -> ✅ 역방향 간선 아이디어
