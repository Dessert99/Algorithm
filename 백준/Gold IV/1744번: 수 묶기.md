[https://www.acmicpc.net/problem/1744](https://www.acmicpc.net/problem/1744)
```javascript
const input = require('fs')
    .readFileSync(process.platform === 'linux' ? '/dev/stdin' : './input.txt')
    .toString()
    .trim()
    .split('\n');

let inputIndex = 0;

// 수열 길이 < 50
const N = Number(input[inputIndex++]);

const positiveArr = [];
const negativeArr = [];
const oneArr = [];
const zeroArr = [];
for (let i = 0; i < N; i++) {
    // 숫자 넣기
    const cur = Number(input[inputIndex++]);

    if (cur < 0) {
        negativeArr.push(cur);
    } else if (cur === 0) {
        zeroArr.push(cur);
    } else if (cur === 1) {
        oneArr.push(cur);
    } else {
        positiveArr.push(cur);
    }
}

positiveArr.sort((a, b) => a - b);
negativeArr.sort((a, b) => b - a);

let answer = 0;

// 양수 더하기
while (positiveArr.length > 1) {
    // 뒤에서 2개 수 꺼내기. 두 개는 무조건 있다.
    const num1 = positiveArr.pop();
    const num2 = positiveArr.pop();

    answer += num1 * num2;
}
// 양수가 홀수였다면 한 개가 남는다.
if (positiveArr.length >= 1) {
    answer += positiveArr.pop();
}
// 1을 전부 더해주기
answer += oneArr.length;

// 음수 더하기
while (negativeArr.length > 1) {
    // 뒤에서 2개 수 꺼내기. 두 개는 무조건 있다.
    const num1 = negativeArr.pop();
    const num2 = negativeArr.pop();

    answer += num1 * num2;
}

// 음수도 홀수였다면 한 개가 남는다
if (negativeArr.length === 1) {
    // 이때, 만약 0이 있다면 곱해서 없앨 수 있다.
    if (zeroArr.length === 0) {
        // 아쉽게도 0이 없다.
        answer += negativeArr.pop();
    }
}

console.log(answer);
```
### 🔗 풀이
1. 🤔 묶는다 -> 묶은 수는 곱한다.

2. 🤔 묶을 때 고려사항들
    1. 0은 안 묶는게 좋다. 혹은, "홀수가 되어 하나 남은 음수"와 묶는다.
    2. 음수끼리 곱해야 한다.
    3. 큰 수끼리 곱한다. (음수는 작은 수 끼리)

3. 🤔 정렬해야할까?
    - 두 수를 곱할 때 크기가 기준이 되니 정렬해야 한다.

4. 🤔 묶을지 안 묶을지 판단하고 구현은 어떻게 할까?
    - 수를 3가지로 나누어 받는다.
        1. 음수
        2. 0
        3. 양수

5. 🤔 예외 조건
    - 1과 -1은 곱하는게 아니라, 그냥 더해야 더 큰 수가 나오는 경우가 있다.
    - 일단 양수 1은 따로 빼서 마지막에 더해준다.
    - 근데 음수 -1은 따로 빼면 안된다. -를 지워주는 용도로 사용하기 때문이다.
