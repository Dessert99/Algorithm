[https://www.acmicpc.net/problem/1197](https://www.acmicpc.net/problem/1197)
```javascript
const input = require('fs')
    .readFileSync(process.platform === 'linux' ? '/dev/stdin' : './input.txt')
    .toString()
    .trim()
    .split('\n');

let inputIndex = 0;

const [V, E] = input[inputIndex++].split(' ').map(Number);

const road = [];
for (let i = 0; i < E; i++) {
    road.push(input[inputIndex++].split(' ').map(Number));
}
// 가중치 오름차순으로 정렬
road.sort((a, b) => a[2] - b[2]);

// MST 가중치
let answer = 0;

// 유니온 파인드 만들기. (0번은 쓰지 않는다.)
const parents = Array.from({ length: V + 1 }, (_, idx) => idx);

function find(x) {
    // 일단 저장
    let root = x;

    // 루트 찾기. 부모가 자기 자신이 아니라면 루트가 아니다.
    while (parents[root] !== root) {
        root = parents[root]; // 하나씩 올라간다.
    }
    // 일단 여기서 루트는 찾았음.

    // 경로 압축
    let cur = x;
    // 지나온 모든 경로를 root로 바꾼다. 현재 노드 cur가 루트를 가리키고 있지 않는다면 계속 반복
    while (cur !== root) {
        let next = parents[cur]; // 지금 노드가 가리키고 있는 부모
        parents[cur] = root; // 부모를 루트로 변경
        cur = next; // 지금 노드를 부모 노드로 변경
    }

    // 최종 노드 리턴
    return root;
}

function uni(u, v, c) {
    // 두 노드의 루트
    const root1 = find(u);
    const root2 = find(v);

    // 루트가 다르다면 한쪽에 붙인다.
    if (root1 !== root2) {
        parents[root2] = root1; // v를 u에 붙이기
        answer += c;
    }
}

// paretns[i] 값들이 모두 1이 되도록 연결한다.
for (let i = 0; i < E; i++) {
    // A,B점이 가중치 C로 연결됨
    const [A, B, C] = road[i];

    // 다르면 붙인다.
    uni(A, B, C);
}

console.log(answer);
```

### 🔗 배운점
1. 크루스칼이란
   > 가장 적은 비용으로 모든 노드를 연결하기 위한 알고리즘. 즉, MST를 만들기 위한 알고리즘  
      ex) 여러 개의 도시를 도로를 이용해 연결하고자 할 때 비용을 최소한으로 하고자 할 때 사용
   - 포인트
     - 사이클이 발생하는지 확인 중요 -> 유니온-파인드에서 `Find`로 사이클 확인한다.
2. 과정
   1. 유니온 파인드에서 만들었던 것처럼 각 노드가 자기 자신을 부모로 가리키는 배열을 만듭니다. (`parent[i] = i`)
   2. 모든 간선(도로) 정보를 가중치 기준으로 오름차순 정렬합니다.
   3. 가중치가 낮은 간선부터 하나씩 꺼내서, 연결하려는 두 정점의 '최상위 부모(루트)'가 같은지 먼저 확인합니다.
      - 부모가 같다면, 이미 연결된 상태이므로 연결하지 않고 넘어갑니다 (사이클 방지).
   4. 두 정점의 최상위 부모가 다르다면, 사이클이 생기지 않으므로 두 정점을 연결하고, 한쪽의 최상위 부모를 다른 쪽 최상위 부모로 갱신합니다
