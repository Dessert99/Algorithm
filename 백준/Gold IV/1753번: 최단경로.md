```javascript
const input = require('fs')
    .readFileSync(process.platform === 'linux' ? '/dev/stdin' : './input.txt')
    .toString()
    .trim()
    .split('\n');

let inputIndex = 0;

// 우선순위 큐
class MinHeap {
    constructor() {
        this.heap = [];
    }

    // 유틸
    swap(i1, i2) {
        [this.heap[i1], this.heap[i2]] = [this.heap[i2], this.heap[i1]];
    }

    getParentIndex(curIndex) {
        return Math.floor((curIndex - 1) / 2);
    }
    getLeftChildIndex(curIndex) {
        return curIndex * 2 + 1;
    }
    getRightChildIndex(curIndex) {
        return curIndex * 2 + 2;
    }

    // 삽입
    insert(value) {
        this.heap.push(value);
        this.heapifyUp();
    }

    heapifyUp() {
        let curIndex = this.heap.length - 1;

        while (curIndex > 0) {
            const parentIndex = this.getParentIndex(curIndex);

            // 종료 조건: 부모보다 크거나 같은 경우
            if (this.heap[parentIndex][1] <= this.heap[curIndex][1]) {
                break;
            }

            this.swap(parentIndex, curIndex);
            curIndex = parentIndex;
        }
    }

    // 삭제
    extractMin() {
        if (this.heap.length === 0) {
            return null;
        }
        if (this.heap.length === 1) {
            return this.heap.pop();
        }

        const min = this.heap[0];

        this.heap[0] = this.heap.pop();

        this.heapifyDown();

        return min;
    }

    heapifyDown() {
        let curIndex = 0;
        const length = this.heap.length;

        while (true) {
            let minIndex = curIndex;
            const leftChildIndex = this.getLeftChildIndex(curIndex);
            const rightChildIndex = this.getRightChildIndex(curIndex);

            if (leftChildIndex < length && this.heap[leftChildIndex][1] < this.heap[minIndex][1]) {
                minIndex = leftChildIndex;
            }
            if (rightChildIndex < length && this.heap[rightChildIndex][1] < this.heap[minIndex][1]) {
                minIndex = rightChildIndex;
            }

            if (curIndex === minIndex) {
                break;
            }

            this.swap(curIndex, minIndex);
            curIndex = minIndex;
        }
    }
}

// V: 정점의 개수(1~2만), E: 간선의 개수(1~30만)
const [V, E] = input[inputIndex++].split(' ').map(Number);

// 시작 정점의 번호
const K = Number(input[inputIndex++]);

// 간선 목록: 인접 리스트. 인덱스 편의성을 위해 크게 만든다.
const adjList = Array.from({ length: V + 1 }, () => []);
for (let i = 0; i < E; i++) {
    // u에서 v로 가는 가중치 w(1~10)인 간선. u !== v
    const [u, v, w] = input[inputIndex++].split(' ').map(Number);

    // 인접 리스트에 들어갈 정보: [도착 정점, 이동 비용]
    adjList[u].push([v, w]);
}

// 가중치 배열. dist[i] = 기준 정점에서 i번째 정점까지 가중치
const dist = Array.from({ length: V + 1 }, () => Infinity);
dist[K] = 0; // 시작 정점 가중치 초기화 필요

// 우선순위 힙
const pq = new MinHeap();
pq.insert([K, 0]); // 초기값 설정 [정점 번호, 그 정점으로 가는데 걸리는 비용]

while (pq.heap.length > 0) {
    const [curV, curCost] = pq.extractMin();

    // 이미 최단 거리로 계산하였다면 무시
    if (dist[curV] < curCost) {
        continue;
    }

    // 현재 정점에서 연결되었는지 확인
    const curLinkedNodes = adjList[curV];

    // 연결된 정점 확인
    for (let j = 0; j < curLinkedNodes.length; j++) {
        // 연결된 정점, 그 정점까지 가는데 비용
        const [linkedV, linkedCost] = curLinkedNodes[j];

        // 해당 정점으로 이동했을 때 비용 (지금까지 누적 비용 + 연결된 정점까지 가는데 비용)
        const nextCost = curCost + linkedCost;

        // 그 정점으로 가는 비용이 더 효율적이라면 dist 갱신
        if (dist[linkedV] > nextCost) {
            dist[linkedV] = nextCost;

            // 더 효율적인 경로이므로 큐에 넣는다.
            pq.insert([linkedV, nextCost]);
        }
    }
}
let answer = [];

for (let i = 1; i < V + 1; i++) {
    if (dist[i] === Infinity) {
        answer.push('INF');
    } else {
        answer.push(dist[i]);
    }
}

console.log(answer.join('\n'));
```
### 🔗 풀이
1. 💡 주의사항
    - 서로 다른 두 정점 사이에 여러 개의 간선이 존재할 수도 있음
    - 간선의 방향이 존재한다.

### 🔗 배운점
1. 인접 행렬? 인접 리스트?
    - 인접 행렬은 $O(n^2)$ 만큼 메모리를 차지한다. 정점 개수가 최대 2만개이니, 총 4억 개의 공간이 필요하다. 메모리 초과가 발생한다.
    - 인접 리스트는 $O(V+E)$ 만큼 메모리를 차지한다. 시간복잡도 또한 연결된 간선만 고려하기 때문에 효율적이다.
    - 이 문제는 인접 리스트로 풀어야 한다.
2. 가중치 배열은 어떤 구조일까?
    - 다익스트라는 "어떤 정점"에서 "다른 정점"에서 가중치이다.
    - 즉, 1차원 배열이다. 시작 정점 K의 초기화가 필요하다.
3. 힙에 어떤 정보를 넣어야 할까?
    - 정점 번호
    - 시작 정점 K에서 큐에서 꺼낸 정점으로 가는데 걸리는 비용

    }
